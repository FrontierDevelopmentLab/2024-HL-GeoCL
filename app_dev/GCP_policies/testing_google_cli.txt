Testing

Create a new VM

gcloud compute instances create india-test-instance-from-cli \
    --zone=us-central1-a \
    --machine-type=e2-micro \
    --image=debian-11-bullseye-v20240709 \
    --image-project=debian-cloud

stop the VM
gcloud compute instances stop india-test-instance-from-cli --zone=us-central1-a


start the VM
gcloud compute instances start india-test-instance-from-cli --zone=us-central1-a


terminate the VM
gcloud compute instances delete india-test-instance-from-cli --zone=us-central1-a
